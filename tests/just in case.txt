// Pressure Model (D-039, LOCKED)

/**
 * Compute-on-read: current effective pressure level at a tile coordinate.
 *
 * Formula (D-039):
 *   currentPressure = max(0, storedLevel - floor((atMs - updatedAtMs) / PRESSURE_DECAY_MS_PER_LEVEL))
 *
 * Returns 0 for unknown tiles or tiles with no pressure state.
 *
 * @param {{ x: number, y: number } | string} coord
 * @param {number} atMs
 * @returns {number}
 */
export function readPressure(coord, atMs) {
  const tile = worldMap.getTile(coord);
  if (!tile) return 0;
  const p = tile.state.pressure;
  if (!p || typeof p.level !== 'number') return 0;
  const decayedLevels = Math.floor((atMs - p.updatedAtMs) / PRESSURE_DECAY_MS_PER_LEVEL);
  return Math.max(0, p.level - decayedLevels);
}
/**
 * Apply a pressure event to a tile (CAST / HOOKSET / CATCH).
 * Writes tile.state.pressure. Emits PRESSURE_APPLIED.
 *
 * @param {{ x: number, y: number } | string} coord
 * @param {'CAST'|'HOOKSET'|'CATCH'} kind
 * @param {number} atMs
 */
export function applyPressureEvent(coord, kind, atMs) {
  const tile = worldMap.getTile(coord);
  if (!tile) return;
  const increment = PRESSURE_INCREMENTS[kind];
  if (increment === undefined) {
    console.warn(`[fishBehavior] applyPressureEvent: unknown kind "${kind}"`);
    return;
  }
  const currentLevel = readPressure(coord, atMs);
  const newLevel = Math.min(MAX_PRESSURE, currentLevel + increment);
  worldMap.mutateTileState(coord, state => ({
    ...state,
    pressure: {
      level:          newLevel,
      updatedAtMs:    atMs,
      lastCastAtMs:   kind === 'CAST'  ? atMs : (state.pressure?.lastCastAtMs  ?? 0),
      lastCatchAtMs:  kind === 'CATCH' ? atMs : (state.pressure?.lastCatchAtMs ?? 0),
    },
  }));
  bus.emit('PRESSURE_APPLIED', { coord, kind, increment, storedLevel: newLevel, atMs });
}

/**
 * Returns the current diurnal period label for a given tournament-clock time.
 *
 * @param {number} atMs
 * @returns {'DAWN'|'DAY'|'DUSK'|'NIGHT'}
 */
function _diurnalPeriod(atMs) {
  const t = atMs % DAY_MS;
  if (t < DAWN_END_MS) return 'DAWN';
  if (t < DAY_END_MS)  return 'DAY';
  if (t < DUSK_END_MS) return 'DUSK';
  return 'NIGHT';
}
/**
 * Returns the diurnal activity multiplier for a species at a given time.
 *
 * @param {object} species
 * @param {number} atMs
 * @returns {number}
 */
function _diurnalMultiplier(species, atMs) {
  return species.diurnal[_diurnalPeriod(atMs)];
}
/**
 * Compute how suitable a tile is for a species [0..1].
 * Based entirely on immutable tile traits — no live state consulted (H-014 safety).
 *
 * @param {object} species
 * @param {object} tile
 * @returns {number}
 */
function _habitatSuitability(species, tile) {
  let score = 0;
  // Depth affinity: 30% weight. Decays linearly over ±3m from preferred depth.
  const depthDiff = Math.abs(tile.traits.depth.minM - species.habitat.depthAffinityM);
  score += Math.max(0, 1 - depthDiff / 3.0) * 0.30;
  // Bottom type: 20% weight.
  if (species.habitat.bottomAffinity.includes(tile.traits.bottom.primary)) {
    score += 0.20;
  }
  // Cover type: 30% weight.
  if (species.habitat.coverAffinity.includes(tile.traits.cover.type)) {
    score += 0.30;
  }
  // Tag affinity: 10% per matching tag (up to 2 tags = 20% max).
  const matchingTags = tile.traits.tags.filter(t =>
    species.habitat.tagAffinity.includes(t)
  );
  score += Math.min(0.20, matchingTags.length * 0.10);
  return Math.min(1.0, score);
}
/**
 * Populate tile occupancy for all registered POIs.
 * Called from onMount when entering TOURNAMENT_ACTIVE.
 * Writes tile.state.occupancy for each tile based on species habitat suitability
 * and structural score from the structureIndex.
 *
 * @param {number} atMs - current tournament clock time
 */
function _populateAllPois(atMs) {
  const poiIds = worldMap.allPoiIds();
  for (const poiId of poiIds) {
    if (!structureIndex.isBuilt(poiId)) continue;
    const candidates = structureIndex.candidatesForPoi(poiId);
    for (const candidate of candidates) {
      const tile = worldMap.getTile(candidate.coord);
      if (!tile) continue;
      // Sum suitability across all species to get total fish potential.
      let totalFishPotential = 0;
      for (const species of Object.values(SPECIES_CATALOG)) {
        const suitability = _habitatSuitability(species, tile);
        totalFishPotential += suitability;
      }
      // Scale by structureScore and RNG to get fishCount.
      // structureScore ∈ [0,1]; totalFishPotential ∈ [0,7] (7 species).
      // Target range: 0–8 fish per tile, weighted by structure.
      const roll = _fishStream.next();
      const fishCount = Math.round(
        totalFishPotential * candidate.structureScore * 3.5 * (0.5 + roll)
      );
      worldMap.mutateTileState(candidate.coord, state => ({
        ...state,
        occupancy: {
          fishCount:          Math.max(0, fishCount),
          fishCountStaleAtMs: atMs,
        },
      }));
    }
  }
}

/**
 * Check if the lure at the given coord has been recently rejected.
 * Returns LURE_REJECTION_PENALTY if recently rejected, 1.0 otherwise.
 *
 * @param {string} coordKey
 * @param {string} lureId
 * @param {number} atMs
 * @returns {number}
 */
function _lureMemoryPenalty(coordKey, lureId, atMs) {
  const memory = _lureMemory.get(coordKey);
  if (!memory) return 1.0;
  const recent = memory.filter(
    m => m.lureId === lureId && (atMs - m.rejectedAtMs) < LURE_MEMORY_DECAY_MS
  );
  return recent.length > 0 ? LURE_REJECTION_PENALTY : 1.0;
}
/**
 * Record a lure rejection at a coord (called by fightLoop on HOOKSET_MISSED or
 * BITE_CANCELLED, then echoed here).
 *
 * @param {{ x: number, y: number }|string} coord
 * @param {string} lureId
 * @param {number} atMs
 */
export function recordLureRejection(coord, lureId, atMs) {
  const tile = worldMap.getTile(coord);
  if (!tile) return;
  const key = typeof coord === 'string' ? coord
    : `${Math.round(tile.coord.x)},${Math.round(tile.coord.y)}`;
  let memory = _lureMemory.get(key);
  if (!memory) {
    memory = [];
    _lureMemory.set(key, memory);
  }
  memory.push({ lureId, rejectedAtMs: atMs });
  // Trim entries older than the decay window to avoid unbounded growth.
  const cutoff = atMs - LURE_MEMORY_DECAY_MS;
  const trimmed = memory.filter(m => m.rejectedAtMs >= cutoff);
  _lureMemory.set(key, trimmed);
}

/**
 * Compute the presentation match score for a lure against a species.
 * Preferred = 1.0, tolerated = 0.50, rejected = 0.05, unknown = 0.40.
 *
 * @param {object} lureSpec
 * @param {object} species
 * @returns {number}
 */
function _presentationMatch(lureSpec, species) {
  const { preferred, tolerated, rejected } = species.presentationPreferences;
  if (preferred.includes(lureSpec.category))  return 1.00;
  if (tolerated.includes(lureSpec.category))  return 0.50;
  if (rejected.includes(lureSpec.category))   return 0.05;
  return 0.40;
}
/**
 * Compute depth match score: how well does the lure's run depth match the
 * species' depth affinity. Score decays over ±4m difference.
 *
 * @param {number} lureDepthM
 * @param {number} fishDepthAffinityM
 * @returns {number}
 */
function _depthMatch(lureDepthM, fishDepthAffinityM) {
  const diff = Math.abs(lureDepthM - fishDepthAffinityM);
  return Math.max(0, 1 - diff / 4.0);
}
/**
 * Resolve the absolute tile coordinate from a cast's landing offset + POI center.
 * Returns null if the POI does not exist.
 *
 * @param {string} poiId
 * @param {{ dx: number, dy: number }} landingOffset
 * @returns {{ x: number, y: number }|null}
 */
function _landingCoord(poiId, landingOffset) {
  const poi = poiGraph.getPoi(poiId);
  if (!poi) return null;
  return {
    x: Math.round(poi.centerCoord.x + landingOffset.dx),
    y: Math.round(poi.centerCoord.y + landingOffset.dy),
  };
}
/**
 * Generate a unique fish instance ID.
 */
let _fishIdCounter = 1;
function _nextFishId() { return `fish_${_fishIdCounter++}`; }
/**
 * Create a live fish instance from a species and cast context.
 *
 * @param {string} speciesId
 * @param {object} species
 * @param {{ x: number, y: number }} coord
 * @param {object} castSpec
 * @param {string} lureId
 * @returns {object}
 */
function _createFishInstance(speciesId, species, coord, castSpec, lureId) {
  const stream = _fishStream;
  const weightKg = Math.max(
    0.05,
    stream.lognormal(species.weight.mu, species.weight.sigma)
  );
  return {
    id:        _nextFishId(),
    speciesId,
    weightKg:  Math.round(weightKg * 1000) / 1000,
    stamina:   species.stamina,
    phase:     'RUNNING',
    coord,
    poiId:     castSpec.poiId,
    lureId,
    lureMemory: [], // D-049 per-instance memory (see also module-level _lureMemory)
  };
}
/**
 * Evaluate whether a cast lands a strike.
 *
 * Process:
 *   1. Apply CAST pressure to landing tile (D-039).
 *   2. Check tile existence and fish occupancy.
 *   3. For each species: compute combined bite probability factoring
 *      spook, pressure, habitat, lure affinity, depth, presentation, diurnal,
 *      lure memory (D-049).
 *   4. Roll in descending probability order; first hit wins.
 *
 * @param {object} castSpec  - the CAST_LANDED event payload
 * @param {string} castSpec.poiId
 * @param {string} castSpec.candidateId
 * @param {string} castSpec.finderTier
 * @param {{ dx: number, dy: number }} castSpec.landing
 * @param {string} castSpec.splashKind
 * @param {number} castSpec.scatterRadius
 * @param {number} castSpec.mitigationFactor
 * @param {number} castSpec.atMs
 * @returns {{ hit: boolean, speciesId?: string, fishInstance?: object }}
 */
export function evaluateStrike(castSpec) {
  const { poiId, landing, atMs } = castSpec;
  // Resolve absolute tile coordinate.
  const absCoord = _landingCoord(poiId, landing);
  if (!absCoord) return { hit: false };
  // Apply cast pressure (D-039: CAST +1).
  applyPressureEvent(absCoord, 'CAST', atMs);
  // Find the tile.
  const tile = worldMap.getTile(absCoord);
  if (!tile) return { hit: false };
  // Must have fish population.
  if (tile.state.occupancy.fishCount === 0) return { hit: false };
  // Environmental modifiers.
  const spookLevel   = castSpookModel.readSpook(absCoord, atMs);
  const pressureLevel = readPressure(absCoord, atMs);
  // Spook factor: full spook → 5% floor (stubborn fish). Zero spook → 1.0.
  const spookFactor    = Math.max(0.05, 1 - spookLevel / castSpookModel.MAX_SPOOK);
  // Pressure factor (D-039): max penalty = 0.6 at full pressure.
  const pressureFactor = 1 - PRESSURE_STRIKE_PENALTY * (pressureLevel / MAX_PRESSURE);
  // Active lure from tournament partition.
  const state        = stateStore.getState();
  const activeTackle = state.tournament?.activeTackle ?? { rods: [], lures: [], bait: [] };
  const activeLure   = activeTackle.lures?.[0];
  if (!activeLure) return { hit: false };
  let lureSpec;
  try { lureSpec = equipment.getLure(activeLure.id); }
  catch { return { hit: false }; }
  // Coord key for lure memory lookup.
  const coordKey = `${absCoord.x},${absCoord.y}`;
  // Build candidates per species.
  const candidates = [];
  for (const [speciesId, species] of Object.entries(SPECIES_CATALOG)) {
    const habitatScore   = _habitatSuitability(species, tile);
    if (habitatScore < 0.12) continue; // species can't meaningfully live here
    const lureAffinity   = lureSpec.speciesAffinity?.[speciesId] ?? 0.50;
    const depthScore     = _depthMatch(
      lureSpec.presentation.runDepthM,
      species.habitat.depthAffinityM
    );
    const presentScore   = _presentationMatch(lureSpec, species);
    const diurnalMult    = _diurnalMultiplier(species, atMs);
    const moodRoll       = 0.5 + _fishStream.next() * species.moodVolatility;
    const memPenalty     = _lureMemoryPenalty(coordKey, activeLure.id, atMs);
    const baseProbability =
      habitatScore * lureAffinity * depthScore * presentScore * diurnalMult * moodRoll * memPenalty;
    const finalP = baseProbability * spookFactor * pressureFactor;
    if (finalP > 0) {
      candidates.push({ speciesId, species, finalP });
    }
  }
  if (candidates.length === 0) return { hit: false };
  // Roll in descending probability order; first hit wins.
  candidates.sort((a, b) => b.finalP - a.finalP);
  for (const c of candidates) {
    if (_fishStream.next() < c.finalP) {
      const fishInstance = _createFishInstance(
        c.speciesId, c.species, absCoord, castSpec, activeLure.id
      );
      return { hit: true, speciesId: c.speciesId, fishInstance };
    }
  }
  return { hit: false };
}

// Bite Timer (D-032, D-033, D-049, D-050)
// ===========================================================================
/**
 * Compute the dynamic hookset window width for a species (D-033, D-050).
 * Baseline 750ms, shrinks by species intelligence (smarter = narrower window).
 * Hard floor: 300ms (D-050, LOCKED).
 *
 * @param {object} species
 * @returns {number} window duration in ms
 */
function _hooksetWindowMs(species) {
  const shrink = species.intelligence * 400; // max shrink = 400ms at intelligence=1
  return Math.max(HOOKSET_WINDOW_FLOOR_MS, Math.round(HOOKSET_WINDOW_BASELINE_MS - shrink));
}
/**
 * Compute dynamic nibble count for a bite sequence (D-032).
 * Factors: species nibbleBand, intelligence (more cautious = more nibbles),
 * diurnal mood (active period = fewer nibbles, fish is committed).
 *
 * @param {object} species
 * @param {number} atMs
 * @returns {number}
 */
function _computeNibbleCount(species, atMs) {
  const base          = _fishStream.int(species.nibbleBand.min, species.nibbleBand.max);
  const intelligenceBonus = Math.floor(species.intelligence * 2 * _fishStream.next()); // 0–2
  const diurnalReduction = _diurnalMultiplier(species, atMs) > 1.2 ? 1 : 0; // active period
  return Math.max(1, base + intelligenceBonus - diurnalReduction);
}
/**
 * Schedule the full bite sequence for a fish that has committed to the lure.
 * Emits BITE_NIBBLE events (N times) then BITE_THUD.
 * Returns a { cancel() } handle; the caller (internal CAST_LANDED handler) stores it.
 *
 * fightLoop subscribes to BITE_THUD to handle the hookset window.
 * fightLoop subscribes to BITE_NIBBLE to install the nibble-trap input blocker.
 *
 * @param {object} fishInstance
 * @param {object} castSpec
 * @returns {{ cancel(): void }}
 */
export function scheduleBite(fishInstance, castSpec) {
  const species       = SPECIES_CATALOG[fishInstance.speciesId];
  const nibbleCount   = _computeNibbleCount(species, castSpec.atMs);
  const windowMs      = _hooksetWindowMs(species);
  const clockHandles  = [];
  let cancelled       = false;
  function cancel() {
    if (cancelled) return;
    cancelled = true;
    for (const h of clockHandles) clock.cancel(h);
    clockHandles.length = 0;
    _activeBites.delete(fishInstance.id);
  }
  // Build nibble schedule.
  let cumulativeMs = 0;
  for (let i = 0; i < nibbleCount; i++) {
    const intervalMs = _fishStream.int(NIBBLE_INTERVAL_MIN_MS, NIBBLE_INTERVAL_MAX_MS);
    cumulativeMs    += intervalMs;
    const fireAtMs   = castSpec.atMs + cumulativeMs;
    const nibbleIdx  = i;
    const h = clock.schedule(cumulativeMs, (atMs) => {
      if (cancelled) return;
      bus.emit('BITE_NIBBLE', {
        fishId:       fishInstance.id,
        speciesId:    fishInstance.speciesId,
        nibbleIndex:  nibbleIdx,
        totalNibbles: nibbleCount,
        poiId:        fishInstance.poiId,
        atMs,
      });
    });
    clockHandles.push(h);
  }
  // BITE_THUD fires after all nibbles plus one final interval.
  const thudInterval = _fishStream.int(NIBBLE_INTERVAL_MIN_MS, NIBBLE_INTERVAL_MAX_MS);
  cumulativeMs      += thudInterval;
  const thudHandle = clock.schedule(cumulativeMs, (atMs) => {
    if (cancelled) return;
    bus.emit('BITE_THUD', {
      fishInstance,
      hooksetWindowMs: windowMs,
      castSpec,
      atMs,
    });
    _activeBites.delete(fishInstance.id);
  });
  clockHandles.push(thudHandle);
  const handle = { cancel };
  _activeBites.set(fishInstance.id, handle);
  return handle;
}
/**
 * Cancel a pending bite sequence by its handle.
 *
 * @param {{ cancel(): void }} handle
 */
export function cancelBite(handle) {
  if (handle && typeof handle.cancel === 'function') {
    handle.cancel();
  }
}

// Fish State Machine — advanceFight (D-031, D-034)
// ===========================================================================
/**
 * Interpolate the pull force curve at a given exhaustion fraction.
 * pullForceCurve has 8 points covering exhaustion 0%→100%.
 *
 * @param {readonly number[]} curve - 8-point pull force array
 * @param {number} exhaustion - [0,1] fraction of stamina exhausted
 * @returns {number} pull force in [0,1]
 */
function _interpolatePullCurve(curve, exhaustion) {
  const idx   = exhaustion * (curve.length - 1);
  const lo    = Math.floor(idx);
  const hi    = Math.min(curve.length - 1, lo + 1);
  const frac  = idx - lo;
  return curve[lo] * (1 - frac) + curve[hi] * frac;
}
/**
 * Advance the fish's fight state machine by one tick.
 * Called by fightLoop on every FIGHT_TICK.
 *
 * Returns updated { stamina, phase, pullForce } for fightLoop to use.
 *
 * @param {{
 *   speciesId: string,
 *   stamina:   number,
 *   phase:     'RUNNING'|'TIRED',
 * }} fightState
 * @param {{
 *   reeling:    boolean,
 *   givingDrag: boolean,
 *   mutex:      boolean,
 * }} inputs
 * @param {number} dt - elapsed time in ms for this tick (typically FIGHT_TICK_MS=60)
 * @returns {{ stamina: number, phase: 'RUNNING'|'TIRED', pullForce: number }}
 */
export function advanceFight(fightState, inputs, dt) {
  const species    = SPECIES_CATALOG[fightState.speciesId];
  const { stamina, phase } = fightState;
  const tickFactor = dt / 60; // normalise to standard 60ms tick
  // Current exhaustion ∈ [0,1].
  const exhaustion = 1 - (stamina / species.stamina);
  const basePull   = _interpolatePullCurve(species.pullForceCurve, exhaustion);
  // Apply fight style variance to pull force.
  let pullModifier = 1.0;
  switch (species.fightStyle) {
    case 'BULLDOG':  pullModifier = 0.95 + _fishStream.next() * 0.10; break; // steady
    case 'RUNNER':   pullModifier = 0.75 + _fishStream.next() * 0.50; break; // variable bursts
    case 'JUMPER':   pullModifier = 0.80 + _fishStream.next() * 0.40; break; // moderate + spikes
    case 'DIVER':    pullModifier = 1.05 + _fishStream.next() * 0.20; break; // high sustained
    case 'THRASHER': pullModifier = 0.50 + _fishStream.next() * 0.70; break; // erratic
    default:         pullModifier = 1.00;
  }
  const pullForce = Math.min(1.0, basePull * pullModifier);
  // Stamina drain/recovery.
  let staminaDelta;
  if (phase === 'RUNNING') {
    const drainRate = inputs.reeling ? 0.015 : 0.007;
    staminaDelta    = -drainRate * tickFactor;
  } else {
    // TIRED: recover, but less if fish is still being reeled.
    const recoverRate = inputs.reeling ? 0.003 : 0.006;
    staminaDelta      = recoverRate * tickFactor;
  }
  const newStamina = Math.max(0, Math.min(species.stamina, stamina + staminaDelta));
  // Phase transitions (D-031).
  let newPhase = phase;
  if (phase === 'RUNNING' && newStamina < 0.15 * species.stamina) {
    newPhase = 'TIRED';
  }
  if (phase === 'TIRED' && newStamina > 0.60 * species.stamina) {
    newPhase = 'RUNNING';
  }
  return { stamina: newStamina, phase: newPhase, pullForce };
}

/**
 * Internal handler for CAST_LANDED events.
 * Evaluates the strike and, on hit, schedules the bite sequence.
 *
 * @param {object} evt - CAST_LANDED event payload
 */
function _onCastLanded(evt) {
  const result = evaluateStrike(evt);
  if (!result.hit) return;
  scheduleBite(result.fishInstance, evt);
}

// Mount Manifest (H-005)

modeRouter.registerMountManifest({
  id:    'fishBehavior',
  modes: ['TOURNAMENT_ACTIVE'],
  onMount(_nextMode, _prevMode) {
    // Create a fresh RNG stream for this tournament (H-015: named 'fish').
    _fishStream = rng.rngStream('fish');
    // Reset lure memory and active bites for the new tournament.
    _lureMemory.clear();
    _activeBites.clear();
    _fishIdCounter = 1;
    // Populate tile occupancy for all registered POIs.
    _populateAllPois(clock.nowMs());
    // Subscribe to CAST_LANDED to drive the strike pipeline.
    _unsubs.push(bus.on('CAST_LANDED', _onCastLanded));
  },
  onUnmount(_prevMode, _nextMode) {
    // Cancel all pending bite sequences.
    for (const handle of _activeBites.values()) handle.cancel();
    _activeBites.clear();
    // Unsubscribe from bus.
    for (const unsub of _unsubs) unsub();

/**
 * AFish Fight Loop — src/casting/fightLoop.js
 *
 * Public API Contract (§9 — CASTING):
 *   (none exported — purely event-driven FSM)
 *
 * Owns the full FIGHT pipeline from BITE_THUD through FIGHT_RESOLVED (D-034..D-036).
 * Consolidated from: hookset.js + tensionModel.js + fight.js (D-066).
 *
 * ── Hookset Trap & Trigger (D-033) ───────────────────────────────────────────
 *
 *   BITE_NIBBLE arrives:
 *     Opens a NIBBLE_WINDOW (NIBBLE_WINDOW_MS). Any INPUT_* event during the window
 *     cancels the bite sequence: BITE_CANCELLED emitted, lure rejection recorded,
 *     bite handle cancelled.
 *
 *   BITE_THUD arrives:
 *     Opens the HOOKSET_WINDOW (duration from fishInstance's species intelligence;
 *     floor 300ms per D-050). Applies HOOKSET pressure to the landing tile (D-039).
 *     Watches for INPUT_ARROW_UP_DOWN edge:
 *       Inside window  → FISH_HOOKED emitted; fight tick starts.
 *       Outside window / timeout / wrong input → HOOKSET_MISSED; lure rejection recorded.
 *
 * ── Fight Tick (D-034) ───────────────────────────────────────────────────────
 *
 *   clock.every(FIGHT_TICK_MS=60) — registered on FISH_HOOKED; cancelled on resolve.
 *   Per tick:
 *     1. Read inputAdapter.isHeld('SPACEBAR') and isHeld('ARROW_DOWN').
 *     2. Detect mutex (both held → idle decay, D-031).
 *     3. Call fishBehavior.advanceFight() for fish stamina/phase.
 *     4. Advance tension via tension model below.
 *     5. Advance landing distance.
 *     6. Check terminal conditions (D-036).
 *     7. Emit coalesced FIGHT_TENSION (D-035).
 *     8. Emit FIGHT_PHASE_CHANGED / FIGHT_THRESHOLD_CROSSED on edge events.
 *
 * ── Tension Model (D-031, D-035, D-036) ──────────────────────────────────────
 *
 *   tension ∈ [0, 1]. Natural progression:
 *
 *   Reeling (SPACEBAR only, no drag):
 *     tension += fishPull × FISH_PULL_REEL_SCALE + REEL_TENSION_ADD
 *
 *   Giving drag (ARROW_DOWN only, no reel):
 *     tension += fishPull × FISH_PULL_DRAG_SCALE − DRAG_TENSION_SUB
 *     (Fish pull softened by giving line; strong fish can still increase tension)
 *
 *   Idle or mutex (D-031 — phase-dependent equilibrium drift):
 *     RUNNING phase: tension += (IDLE_RUNNING_EQ − tension) × IDLE_APPROACH_RATE
 *     TIRED phase:   tension += (IDLE_TIRED_EQ  − tension) × IDLE_APPROACH_RATE
 *
 *   Landing distance:
 *     Decreases by REEL_SPEED when reeling.
 *     Increases by FISH_RUN_SPEED when fish is RUNNING and not being reeled.
 *
 * ── Failure Modes (D-036) ────────────────────────────────────────────────────
 *
 *   tension >= 1.0 → FIGHT_THRESHOLD_CROSSED(SNAP) then FIGHT_RESOLVED(LINE_SNAPPED)
 *   tension <= 0.0 for SLACK_GRACE_MS(1500ms) → FIGHT_THRESHOLD_CROSSED(SLACK_LOST)
 *                                                then FIGHT_RESOLVED(HOOK_SHAKEN)
 *   landingDistance <= LANDED_THRESHOLD → FIGHT_RESOLVED(FISH_LANDED)
 *
 * ── Event Channels (D-035) ───────────────────────────────────────────────────
 *
 *   FIGHT_TENSION        { tension, trend, atMs, rampToMs }
 *     Coalesced: only emitted when |Δtension| > TENSION_EMIT_DELTA (0.02)
 *                OR at least TENSION_EMIT_MAX_GAP_MS (250ms) since last emit.
 *
 *   FIGHT_PHASE_CHANGED  { phase, prevPhase, atMs }            — edge only
 *   FIGHT_THRESHOLD_CROSSED { threshold, atMs }                — edge only
 *     Thresholds: SLACK_DANGER (tension < 0.10), SNAP_DANGER (tension > 0.85),
 *                 SNAP (tension >= 1.0), SLACK_LOST (tension 0 for SLACK_GRACE_MS)
 *
 *   FIGHT_RESOLVED       { outcome, fishInstance, weightKg, atMs }
 *     outcome: 'FISH_LANDED' | 'LINE_SNAPPED' | 'HOOK_SHAKEN'
 *     — terminal; fight tick is cancelled immediately before emit (H-005, H-011).
 *
 *   FISH_HOOKED          { fishInstance, atMs }  — emitted on successful hookset
 *   HOOKSET_MISSED       { speciesId, poiId, atMs }
 *   BITE_CANCELLED       { reason, speciesId, poiId, lureId, atMs }
 *
 * H-005 note: _tickHandle, _hooksetTimeoutHandle, and all _unsubs are cancelled
 *   in onUnmount. If a fight is in-progress during mode transition, it is resolved
 *   as HOOK_SHAKEN (tournament ending mid-fight).
 * H-011 note: FIGHT_TICK_MS = 60ms with D-035 delta-gate prevents bus flooding.
 * H-014 note: No direct import of fishFinder.js.

import * as clock        from '../core/clock.js';
import * as inputAdapter from '../core/inputAdapter.js';
import * as stateStore   from '../core/stateStore.js';
import * as modeRouter   from '../core/modeRouter.js';

// ===========================================================================

const FIGHT_TICK_MS = 60;
/** Tension-at-zero grace period before HOOK_SHAKEN (D-036). */
const SLACK_GRACE_MS = 1_500;
// --- Tension model ---
/** Fraction of fish pull force added to tension per tick when reeling. */
const FISH_PULL_REEL_SCALE = 0.65;
/** Fraction of fish pull force added to tension per tick when giving drag. */
const FISH_PULL_DRAG_SCALE = 0.20;
/** Fixed tension added per tick when reeling (line tightening). */
const REEL_TENSION_ADD = 0.018;
/** Fixed tension removed per tick when giving drag (releasing line). */
const DRAG_TENSION_SUB = 0.040;
/** Running phase idle equilibrium tension (D-031). */
const IDLE_RUNNING_EQ = 0.62;
/** Tired phase idle equilibrium tension (D-031). */
const IDLE_TIRED_EQ = 0.22;
/** Per-tick fraction of the gap to equilibrium that is closed (D-031). */
const IDLE_APPROACH_RATE = 0.10;
// --- Landing distance ---
/** Starting fight distance from the boat in tiles. */
const MAX_LANDING_DISTANCE = 20.0;
/** Landing distance at which FISH_LANDED is triggered. */
const LANDED_THRESHOLD = 0.5;
/** Tiles retrieved per tick when reeling. */
const REEL_SPEED = 0.20;
/** Tiles escaped per tick when fish is RUNNING and angler is not reeling. */
const FISH_RUN_SPEED = 0.12;
// --- Coalesced emission gates (D-035) ---
/** Minimum |Δtension| to trigger a FIGHT_TENSION emit. */
const TENSION_EMIT_DELTA = 0.02;
/** Maximum ms between FIGHT_TENSION emits regardless of delta. */

const SLACK_DANGER_THRESHOLD = 0.10;

/**
 * Resolve the current fight with the given outcome.
 * Cleans up the tick, dispatches SCAN_UNLOCKED, and emits FIGHT_RESOLVED.
 *
 * @param {'FISH_LANDED'|'LINE_SNAPPED'|'HOOK_SHAKEN'} outcome
 * @param {number} atMs

function _resolveFight(outcome, atMs) {
  if (!_fight) return;
  const fishInstance = _fight.fishInstance;
  _clearFight();
  // Apply CATCH pressure if fish was successfully landed (D-039).
  if (outcome === 'FISH_LANDED') {
    fishBehavior.applyPressureEvent(fishInstance.coord, 'CATCH', atMs);

  // Record lure rejection if fish was lost (helps inform lure memory, D-049).

    fishBehavior.recordLureRejection(fishInstance.coord, fishInstance.lureId, atMs);

  stateStore.dispatch({ type: 'SCAN_UNLOCKED', payload: {} });

    fishInstance,
    weightKg: fishInstance.weightKg,

function _onBiteNibble(evt) {

  if (_fight) return;
  // Clear any prior nibble window.
  _clearNibbleWindow();
  // Block any input during the NIBBLE_WINDOW.
  _nibbleInputUnsub = bus.on('INPUT', (inputEvt) => {
    // Any key press during the NIBBLE_WINDOW cancels the bite.
    _clearNibbleWindow();
    _clearHooksetWindow();
    // Cancel the pending bite sequence if it exists.
    // (fishBehavior's bite sequence is keyed by fishId — we signal via BITE_CANCELLED.)
    const activeLure = stateStore.getState().tournament?.activeTackle?.lures?.[0];
    bus.emit('BITE_CANCELLED', {
      reason:    'INPUT_DURING_NIBBLE_WINDOW',
      speciesId: evt.speciesId,

      lureId:    activeLure?.id ?? null,
      atMs:      inputEvt.atMs,

  _nibbleWindowHandle = clock.schedule(NIBBLE_WINDOW_MS, (_atMs) => {
    _clearNibbleWindow();
    // Window expired without input — bite sequence continues to BITE_THUD.
  });
}

/**
 * Handle an incoming BITE_THUD event.
 * Clears any active nibble trap, applies HOOKSET pressure, then opens the
 * HOOKSET_WINDOW watching for INPUT_ARROW_UP_DOWN.
 *
 * @param {object} evt - BITE_THUD event payload
 */
function _onBiteThud(evt) {
  // Ignore if a fight is already active.
  if (_fight) return;
  // Nibble window no longer relevant.
  _clearNibbleWindow();
  // Apply HOOKSET pressure (D-039: HOOKSET +1).
  fishBehavior.applyPressureEvent(evt.fishInstance.coord, 'HOOKSET', evt.atMs);
  _pendingFishInstance = evt.fishInstance;
  // Subscribe to the hookset input edge (D-033: ARROW_UP key).
  // We listen for the _DOWN edge so it fires the moment the key is pressed.
  _hooksetInputUnsub = bus.on('INPUT_ARROW_UP_DOWN', (inputEvt) => {
    _clearHooksetWindow();
    _startFight(_pendingFishInstance, inputEvt.atMs);
    _pendingFishInstance = null;
  });
  // Also subscribe to any OTHER input (wrong key = miss the hookset).
  // We use a separate short-circuit: any INPUT event that is NOT ARROW_UP_DOWN
  // triggers a miss. We detect this by also subscribing to INPUT and checking.
  const _anyInputUnsub = bus.on('INPUT', (inputEvt) => {
    // If ARROW_UP_DOWN already cleared the window, this fires but _hooksetInputUnsub is null.
    if (!_hooksetInputUnsub) { _anyInputUnsub(); return; }
    if (inputEvt.type !== 'ARROW_UP') {
      // Wrong key during hookset window → miss.
      _anyInputUnsub();
      _clearHooksetWindow();
      _onHooksetMiss(_pendingFishInstance, inputEvt.atMs);
      _pendingFishInstance = null;
    }
  });
  // Hookset window timeout (D-033: missed if window expires).
  _hooksetTimeoutHandle = clock.schedule(evt.hooksetWindowMs, (atMs) => {
    _anyInputUnsub(); // ensure cleaned up
    if (!_pendingFishInstance) return; // already resolved
    _clearHooksetWindow();
    _onHooksetMiss(_pendingFishInstance, atMs);
    _pendingFishInstance = null;
  });
}

    speciesId: fishInstance.speciesId,
    poiId:     fishInstance.poiId,

  // Dispatch SCAN_LOCKED (D-043: casting/fighting are mutually exclusive with scan).
  stateStore.dispatch({ type: 'SCAN_LOCKED', payload: {} });
  _fight = {
    fishInstance,
    tension:            0.40,   // comfortable starting tension
    prevTension:        0.40,
    phase:              'RUNNING',
    stamina:            fishInstance.stamina,

    lastEmitAtMs:       atMs,

    thresholds: {
      snapDanger:  false,
      slackDanger: false,

  bus.emit('FISH_HOOKED', { fishInstance, atMs });
  bus.emit('FIGHT_PHASE_CHANGED', { phase: 'RUNNING', prevPhase: null, atMs });
  // Register the fight tick (H-011: 60ms cadence, H-005: cancelled on resolve).
  _tickHandle = clock.every(FIGHT_TICK_MS, _onFightTick);
}
// ===========================================================================
// Fight Tick (D-034)
// ===========================================================================
/**
 * Core fight tick — called every FIGHT_TICK_MS (60ms) by clock.every().
 *
 * @param {number} atMs
 */
function _onFightTick(atMs) {
  if (!_fight) {
    // Defensive: tick should not fire without active fight, but cancel gracefully.
    _clearFight();
    return;
  }
  const f          = _fight;
  const dt         = FIGHT_TICK_MS;
  const tickFactor = dt / 60;
  // ── 1. Read inputs ──────────────────────────────────────────────────────
  const spacebar   = inputAdapter.isHeld('SPACEBAR');
  const arrowDown  = inputAdapter.isHeld('ARROW_DOWN');
  const mutex      = spacebar && arrowDown; // D-031
  const reeling    = spacebar && !mutex;
  const givingDrag = arrowDown && !mutex;
  // ── 2. Advance fish state machine (D-034) ───────────────────────────────
  const prevPhase   = f.phase;
  const fishResult  = fishBehavior.advanceFight(
    { speciesId: f.fishInstance.speciesId, stamina: f.stamina, phase: f.phase },
    { reeling, givingDrag, mutex },
    dt
  );
  f.stamina = fishResult.stamina;
  // Detect phase change (D-035 FIGHT_PHASE_CHANGED edge).
  if (fishResult.phase !== prevPhase) {
    f.phase = fishResult.phase;
    bus.emit('FIGHT_PHASE_CHANGED', {
      phase:     f.phase,
      prevPhase,
      atMs,
    });
  } else {
    f.phase = fishResult.phase;
  }
  // ── 3. Advance tension ─────────────────────────────────────────────────
  const prevTension = f.tension;
  let delta = 0;
  if (reeling) {
    // Reeling: line tightens from both player pull and fish resistance.
    delta += REEL_TENSION_ADD * tickFactor;
    delta += fishResult.pullForce * FISH_PULL_REEL_SCALE * tickFactor;
  } else if (givingDrag) {
    // Giving drag: releases tension but fish still fights somewhat.
    delta -= DRAG_TENSION_SUB * tickFactor;
    delta += fishResult.pullForce * FISH_PULL_DRAG_SCALE * tickFactor;
  } else {
    // Idle or mutex: drift toward phase equilibrium (D-031).
    const equilibrium = f.phase === 'RUNNING' ? IDLE_RUNNING_EQ : IDLE_TIRED_EQ;
    delta += (equilibrium - f.tension) * IDLE_APPROACH_RATE * tickFactor;
  }
  f.tension = Math.max(0, Math.min(1.0, f.tension + delta));
  // ── 4. Advance landing distance ─────────────────────────────────────────
  if (reeling) {
    f.landingDistance = Math.max(0, f.landingDistance - REEL_SPEED * tickFactor);
  } else if (f.phase === 'RUNNING' && !givingDrag) {
    // Fish runs when not being reeled in Running phase.
    f.landingDistance = Math.min(
      MAX_LANDING_DISTANCE,
      f.landingDistance + FISH_RUN_SPEED * tickFactor
    );
  }
  // ── 5. Check terminal conditions (D-036) ────────────────────────────────
  // FISH_LANDED.
  if (f.landingDistance <= LANDED_THRESHOLD) {
    _resolveFight('FISH_LANDED', atMs);
    return;
  }
  // LINE_SNAPPED.
  if (f.tension >= 1.0) {
    bus.emit('FIGHT_THRESHOLD_CROSSED', { threshold: 'SNAP', atMs });
    _resolveFight('LINE_SNAPPED', atMs);
    return;
  }
  // HOOK_SHAKEN (slack grace, D-036).
  if (f.tension <= 0) {
    if (f.slackStartMs === null) {
      f.slackStartMs = atMs;
    } else if (atMs - f.slackStartMs >= SLACK_GRACE_MS) {
      bus.emit('FIGHT_THRESHOLD_CROSSED', { threshold: 'SLACK_LOST', atMs });
      _resolveFight('HOOK_SHAKEN', atMs);
      return;
    }
  } else {
    f.slackStartMs = null;
  }
  // ── 6. Emit threshold crossings (edge events, D-035) ────────────────────
  // SNAP_DANGER: entering the danger zone.
  if (f.tension > SNAP_DANGER_THRESHOLD && !f.thresholds.snapDanger) {
    f.thresholds.snapDanger = true;
    bus.emit('FIGHT_THRESHOLD_CROSSED', { threshold: 'SNAP_DANGER', atMs });
  } else if (f.tension <= SNAP_DANGER_THRESHOLD && f.thresholds.snapDanger) {
    f.thresholds.snapDanger = false;
    // No event on leaving — audio handles decay naturally.
  }
  // SLACK_DANGER: entering the slack danger zone.
  if (f.tension < SLACK_DANGER_THRESHOLD && !f.thresholds.slackDanger) {
    f.thresholds.slackDanger = true;
    bus.emit('FIGHT_THRESHOLD_CROSSED', { threshold: 'SLACK_DANGER', atMs });
  } else if (f.tension >= SLACK_DANGER_THRESHOLD && f.thresholds.slackDanger) {
    f.thresholds.slackDanger = false;
  }
  // ── 7. Coalesced FIGHT_TENSION emission (D-035) ─────────────────────────
  const tensionDelta    = Math.abs(f.tension - f.lastEmitTension);
  const timeSinceEmit   = atMs - f.lastEmitAtMs;
  const shouldEmit      =
    tensionDelta > TENSION_EMIT_DELTA || timeSinceEmit >= TENSION_EMIT_MAX_GAP_MS;
  if (shouldEmit) {
    const trend =
      f.tension > f.lastEmitTension ? 'RISING' :
      f.tension < f.lastEmitTension ? 'FALLING' : 'STEADY';
    bus.emit('FIGHT_TENSION', {
      tension:   f.tension,
      trend,
      atMs,
      rampToMs:  atMs + FIGHT_TICK_MS, // next tick target time for audio ramp (D-065)
    });
    f.lastEmitTension = f.tension;
    f.lastEmitAtMs    = atMs;
  }
  f.prevTension = prevTension;
}
// ===========================================================================
// Mount Manifest (H-005)
// ===========================================================================
modeRouter.registerMountManifest({
  id:    'fightLoop',
  modes: ['TOURNAMENT_ACTIVE'],
  onMount(_nextMode, _prevMode) {
    // Subscribe to bite events.
    _unsubs.push(bus.on('BITE_NIBBLE', _onBiteNibble));
    _unsubs.push(bus.on('BITE_THUD',   _onBiteThud));
  },
  onUnmount(_prevMode, _nextMode) {
    // If a fight is active, resolve as HOOK_SHAKEN (mode change mid-fight).
    if (_fight) {
      _resolveFight('HOOK_SHAKEN', clock.nowMs());
    }
    // Clear all pending windows.
    _clearNibbleWindow();
    _clearHooksetWindow();
    _clearFight();
    // Unsubscribe from persistent bus listeners.
    for (const unsub of _unsubs) unsub();
    _unsubs.length = 0;
  },
});
